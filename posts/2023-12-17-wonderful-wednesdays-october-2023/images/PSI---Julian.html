SITE04 swapped values

For each research site I created a histogram comparing answer distributions against all other sites. I observed that SITE04 had the most “opposite” answer distribution compared to the others. Considering this, I suspect that SITE04 may have swapped answer scales.

# melt data
testData6 <- testData
names(testData6)[6:9] <- paste("Question", 1:4, sep = "")
testData4 <- melt(testData6, 
                  id.vars = c("site", "id", "trt", "sex", "age"), 
                  measure.vars = paste("Question", 1:4, sep = ""), 
                  variable.name = "question", 
                  value.name = "response")

# create plot
printPlot <- function(test, title){
  p1 <- ggplot(test,
               aes(x = response,
                   fill = site)) +
      geom_histogram(aes(y = ..density..), binwidth = 1, show.legend = F) +
      facet_grid(site ~ question) +
      scale_fill_viridis(discrete = T) +
      theme_minimal() +
      theme(axis.title = element_text(face = "bold"),
            plot.title = element_text(hjust = 0.5),
            strip.text.y = element_text(angle = 0))  +
      labs(title = "",
           y = "Density",
           x = "Response")
  
  print(p1)
}  

# split table
compareSites <- function(x){
  setDT(testData4)
  testData4With <- testData4[site == x,]
  testData4Without <- testData4[site != x,]
  testData4Without$site <- "Other Sites"
  testData5 <- rbind(testData4Without, testData4With)
  printPlot(testData5, x)
}

# # print all plots
sites <- unique(testData$site)
for(i in sites){
compareSites(i)
}

compareSites("SITE04")
